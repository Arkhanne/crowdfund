```Ruby
require_relative 'movie'

describe Movie do
  before do
    $strout = StringIO.new
    @initial_rank = 10
    @movie = Movie.new("goonies", initial_rank)
  end
  
  it 'has a capitalized title' do
    expect(@movie.title).to eql("Goonies")
  end
  
  context "created with a default rank" do
```

## Custom Iterators
```Ruby
def once
  puts "Before yield"
  yield
  puts "After yield"
end

once { puts "Running your block" }
```

```Ruby
def three_times
  puts "Ready"
  yield(1)
  puts "Set"
  yield 2
  puts "Go!"
  yield 3
end

three_times { |number| puts number" }
```

```Ruby
def compute
  if block_given?
    puts yield
  else
    puts "Does not compute!"
  end
end

compute { 3.14 }
```

```Ruby
numbers = (1..10).to_a
puts numbers.select { |n| n.even? }

# Same with iterators
numbers = (1..10).to_a

def my_select(array)
  results = []
  array.each do |element|
    results << element if yield(element)
  end
  results
end

puts my_select(numbers) { |n| n.even? }
```
